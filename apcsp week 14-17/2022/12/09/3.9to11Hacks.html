import random

def RandomNumGen(x):
    RandomNumGenList = []
    
    while x > 0:
        RandomNumGenList.append(random.randint(1,20))
        x -= 1

    print(RandomNumGenList)
    return max(RandomNumGenList)

RandomNumGen(3)
